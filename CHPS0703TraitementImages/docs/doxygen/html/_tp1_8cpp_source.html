730 void filtreGaussien(int kernelSize = 5, double sigma = 1.0) {
731 if (kernelSize < 1 || kernelSize % 2 == 0) {
732 throw std::runtime_error("taille noyau impair");
733 }
734 if (sigma <= 0.0) {
735 throw std::runtime_error("sigma positif");
736 }
737
738 const int radius = kernelSize / 2;
739 const double sigma2 = 2.0 * sigma * sigma;
740
741 // NOTE: allocation dynamique du noyau gaussien 2D
742 double** kernel = new double*[kernelSize];
743 for (int i = 0; i < kernelSize; i++) {
744 kernel[i] = new double[kernelSize];
745 }
746
747 double kernelSum = 0.0;
748
749 for (int dy = -radius; dy <= radius; dy++) {
750 for (int dx = -radius; dx <= radius; dx++) {
751 double dist2 = dx * dx + dy * dy;
752 double value = std::exp(-dist2 / sigma2);
753 kernel[dy + radius][dx + radius] = value;
754 kernelSum += value;
755 }
756 }
757
758 // NOTE: normalisation du noyau (somme = 1)
759 for (int dy = 0; dy < kernelSize; dy++) {
760 for (int dx = 0; dx < kernelSize; dx++) {
761 kernel[dy][dx] /= kernelSum;
762 }
763 }
764
765 double** temp = createTempCopy();
766
767 for (int y = 0; y < height; y++) {
768 for (int x = 0; x < width; x++) {
769 for (int c = 0; c < colors; c++) {
770 double sum = 0.0;
771
772 for (int dy = -radius; dy <= radius; dy++) {
773 for (int dx = -radius; dx <= radius; dx++) {
774 int ny = y + dy;
775 int nx = x + dx;
776
777 if (ny >= 0 && ny < height && nx >= 0 && nx < width) {
778 double weight = kernel[dy + radius][dx + radius];
779 sum += weight * temp[ny][nx * colors + c];
780 }
781 }
782 }
783
784 data[y][x * colors + c] = clamp(sum, 0.0, 255.0);
785 }
786 }
787 }
788
789 // NOTE: libération du noyau
790 for (int i = 0; i < kernelSize; i++) {
791 delete[] kernel[i];
792 }
793 delete[] kernel;
794
795 freeMemory(temp);
796 }

830 int ny = y + dy;
831 int nx = x + dx;

992 // NOTE: étape 1 - lissage gaussien
993 filtreGaussien(5, 1.4);
994
995 double** temp = createTempCopy();
996 double** gradient = new double*[height];
997 double** direction = new double*[height];

999 for (int i = 0; i < height; i++) {
1000 gradient[i] = new double[width * colors];

1002 }
1003
1004 // NOTE: étape 2 - calcul du gradient avec Sobel
1005 const int sobelX[3][3] = {{-1,0,+1},{-2,0,+2},{-1,0,+1}};
1006 const int sobelY[3][3] = {{-1,-2,-1},{0,0,0},{+1,+2,+1}};
1007
1008 for (int y = 1; y < height - 1; y++) {
1009 for (int x = 1; x < width - 1; x++) {
1010 for (int c = 0; c < colors; c++) {
1011 double gx = 0.0, gy = 0.0;
1012
1013 for (int dy = -1; dy <= 1; dy++) {
1014 for (int dx = -1; dx <= 1; dx++) {
1015 double val = temp[y + dy][(x + dx) * colors + c];
1016 gx += val * sobelX[dy + 1][dx + 1];
1017 gy += val * sobelY[dy + 1][dx + 1];
1018 }
1019 }
1020
1021 gradient[y][x * colors + c] = std::sqrt(gx * gx + gy * gy);
1022 direction[y][x * colors + c] = std::atan2(gy, gx);
1023 }
1024 }

1027 // NOTE: étape 3 - suppression des non-maximums
1028 for (int y = 1; y < height - 1; y++) {
1029 for (int x = 1; x < width - 1; x++) {
1030 for (int c = 0; c < colors; c++) {
1031 double angle = direction[y][x * colors + c];
1032 double mag = gradient[y][x * colors + c];
1033
1034 // NOTE: quantification de la direction (0, 45, 90, 135 degrés)
1035 angle = angle * 180.0 / 3.14159265;
1036 if (angle < 0) angle += 180.0;
1037
1038 double neighbor1 = 0.0, neighbor2 = 0.0;
1039
1040 if ((angle >= 0 && angle < 22.5) || (angle >= 157.5 && angle <= 180)) {

1051 neighbor2 = gradient[y + 1][(x + 1) * colors + c];

1116 void filtreBilateral(int kernelSize = 5, double sigmaSpatial = 50.0, double sigmaRange = 50.0) {
1117 if (kernelSize < 1 || kernelSize % 2 == 0) {
1118 throw std::runtime_error("taille noyau impair");
1119 }
1120 if (sigmaSpatial <= 0.0 || sigmaRange <= 0.0) {
1121 throw std::runtime_error("sigma positif");
1122 }
1123
1124 const int radius = kernelSize / 2;
1125 const double sigmaS2 = 2.0 * sigmaSpatial * sigmaSpatial;
1126 const double sigmaR2 = 2.0 * sigmaRange * sigmaRange;
1127 double** temp = createTempCopy();
1128
1129 for (int y = 0; y < height; y++) {
1130 for (int x = 0; x < width; x++) {
1131 for (int c = 0; c < colors; c++) {
1132 double centerVal = temp[y][x * colors + c];
1133 double sum = 0.0;
1134 double weightSum = 0.0;
1135
1136 for (int dy = -radius; dy <= radius; dy++) {
1137 for (int dx = -radius; dx <= radius; dx++) {
1138 int ny = y + dy;
1139 int nx = x + dx;
1140
1141 if (ny >= 0 && ny < height && nx >= 0 && nx < width) {
1142 double neighborVal = temp[ny][nx * colors + c];
1143
1144 // NOTE: pondération spatiale (distance géométrique)
1145 double spatialDist2 = dx * dx + dy * dy;
1146 double spatialWeight = std::exp(-spatialDist2 / sigmaS2);
1147
1148 // NOTE: pondération d'intensité (différence de valeur)
1149 double valueDiff = centerVal - neighborVal;
1150 double rangeWeight = std::exp(-(valueDiff * valueDiff) / sigmaR2);
1151
1152 // NOTE: poids combiné (produit des deux gaussiennes)
1153 double weight = spatialWeight * rangeWeight;
1154
1155 sum += weight * neighborVal;
1156 weightSum += weight;
1157 }
1158 }
1159 }
1160
1161 data[y][x * colors + c] = (weightSum > 0.0) ? (sum / weightSum) : centerVal;
1162 }
1163 }
1164 }
1165
1166 freeMemory(temp);
1167 }
